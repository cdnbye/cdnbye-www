# Project Setup

# For iOS only

To install SDK into your Xcode project using CocoaPods, specify it in your Podfile:

target 'TargetName' do
# Uncomment the next line if you're using Swift
# use_frameworks!
    pod 'SwarmCloudSDK', '= 2.6.5'
end

Then, run the following command:

$ pod install

If can not find SwarmCloudSDK in repo, run command:

$ pod repo update

Update SDK if needed:

pod update SwarmCloudSDK --verbose --no-repo-update

# For iOS/tvOS/macOS

# Install dependencies

# CocoaPods

To install dependencies into your Xcode project using CocoaPods, specify it in your Podfile:

target 'TargetName' do
# Uncomment the next line if you're using Swift
# use_frameworks!
    pod 'SocketRocket', '~> 0.6'
    pod 'CocoaLumberjack', '~> 3.7'
    pod "GCDWebServer", :git => 'https://github.com/swisspol/GCDWebServer.git'
    pod 'CocoaAsyncSocket', '~> 7.6'
    pod 'PINCache', '~> 3.0'
end

Then, run the following command:

$ pod install

# Carthage

Add the dependencies to the Cartfile:

github "facebook/SocketRocket" ~> 0.6
github "CocoaLumberjack/CocoaLumberjack" ~> 3.7
github "pinterest/PINCache" ~> 3.0
github "https://github.com/swisspol/GCDWebServer.git" "master"
github "robbiehanson/CocoaAsyncSocket" ~> 7.6

Then run:

carthage update --use-xcframeworks --cache-builds

This will build each one or download a pre-compiled XCFramework.
On your application targets’ General settings tab, in the Frameworks, Libraries, and Embedded Content section, drag and drop each XCFramework you want to use from the Carthage/Build folder.

# Copy the framework

Unzip files, then drag and drop SwarmCloudSDK.xcframework and WebRTC.xcframework into your Xcode project on your application targets’ General settings tab, in the Frameworks, Libraries, and Embedded Content section.

# Integration

In order to allow the loading of distributed content via the local proxy, enable loading data from HTTP in your app by opening your info.plist file as source code and adding the following values below the tag:

<key>NSAppTransportSecurity</key>
<dict>
    <key>NSAllowsArbitraryLoads</key>
    <true/>
</dict>

# Include

Import SwarmCloudSDK in AppDelegate.m:

import SwarmCloudSDK

#import <SwarmCloudSDK/SwarmCloudSDK.h>

# Initialize SWCP2pEngine

Initialize SWCP2pEngine in AppDelegate.m:

func application(_ application: UIApplication, didFinishLaunchingWithOptions launchOptions: [UIApplication.LaunchOptionsKey: Any]?) -> Bool {
    let config = SWCP2pConfig.defaultConfiguration()
    config.announceLocation = SWCAnnounceLocation.china      // Set HongKong or USA if you changed zone
    SWCP2pEngine.sharedInstance().start(token: YOUR_TOKEN, p2pConfig: config)
    return true
}

- (BOOL)application:(UIApplication *)application didFinishLaunchingWithOptions:(NSDictionary *)launchOptions
{
    SWCP2pConfig *config = [SWCP2pConfig defaultConfiguration];
    config.announceLocation = SWCAnnounceLocationChina;    // Set HongKong or USA if you changed zone
    [[SWCP2pEngine sharedInstance] startWithToken:YOUR_TOKEN andP2pConfig:config];
    return YES;
}

Where YOUR_TOKEN is your Customer ID. Please replace it by your own token obtained from console, click here for more information.

# Usage

When initializing an AVPlayer (or any other video player) instance, before passing it a URL, pass that URL through P2P Engine:

let orginalUrl = URL.init(string: "https://your_stream.m3u8")
let parsedUrl = SWCP2pEngine.sharedInstance().parse(streamURL: orginalUrl)
_player = AVPlayer.init(url: parsedUrl)

NSURL *originalUrl = [NSURL URLWithString:@"https://your_stream.m3u8"];
NSURL *parsedUrl = [[SWCP2pEngine sharedInstance] parseStreamURL:originalUrl];
_player = [[AVPlayer alloc] initWithURL:parsedUrl];

That’s it! SDK should now be integrated into your app.

Simulator may not work as expected, please test on real device.
